## Code

```rust
pub struct Document {
    content: String,
    has_utf8_bom: bool,
}

impl Document {
    pub fn from_bytes(bytes: &[u8]) -> Result<Document, EncodingError> {
        if bytes.starts_with(&[0xFF, 0xFE]) { return Err(EncodingError::Utf16Le); }
        let content = std::str::from_utf8(bytes).map_err(|_| EncodingError::InvalidUtf8)?.to_string();
        Ok(Document { content, has_utf8_bom: false })
    }
}
```
